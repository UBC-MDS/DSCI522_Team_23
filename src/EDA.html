import altair as alt

fixed_acidity = (
    alt.Chart(wine)
    .mark_bar()
    .encode(
        x=alt.X("quality"),
        y=alt.Y("mean(fixed acidity)", scale=alt.Scale(domain=(6, 8.5))),
        color=alt.Color("quality", title="Wine Grade"),
